
    public static int Fibonacci(int n)
    {
        if (n <= 1)
            return n;

        int previousNumber = 0;
        int currentNumber = 1;
        int result = 0;

        for (int i = 2; i <= n; i++)
        {
            result = previousNumber + currentNumber;
            previousNumber = currentNumber;
            currentNumber = result;
        }

        return result;
    }
